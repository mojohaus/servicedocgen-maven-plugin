56  /**
57   * {@link AbstractMojo Maven Plugin} to automatically generate documentation for services of the current project.
58   * <ol>
59   * <li>Scans the current projects source code for (JAX-RS annotated) services that match the RegEx configured by
60   * <code>classnameRegex</code>.</li>
61   * <li>Analyzes the services from source-code (extract JavaDoc, etc.) and byte-code (resolve generic parameters, etc.)
62   * and create intermediate meta-data as {@link ServicesDescriptor}.</li>
63   * <li>Generates documentation from the collected meta-data (by default as HTML from a velocity template shipped with
64   * this plugin but can be overridden via configuration parameters).</li>

248     private List<JavaClass> scanServices( JavaProjectBuilder builder )
249         throws IOException
250     {
251         List<JavaClass> serviceClasses = new ArrayList<JavaClass>();
252         for ( String sourceDir : this.project.getCompileSourceRoots() )
253         {
254             File sourceFolder = new File( sourceDir );
255             if ( sourceFolder.isDirectory() )
256             {
257                 builder.addSourceFolder( sourceFolder );
258                 if ( this.serviceClassName == null )
259                 {
260                     scanJavaFilesRecursive( sourceFolder, builder, serviceClasses );
261                 }
262             }
263         }
264         if ( this.serviceClassName != null )
265         {
266             JavaClass type = builder.getClassByName( this.serviceClassName );
267             boolean isService = isServiceClass( type );
268             if ( isService )
269             {
270                 serviceClasses.add( type );
271             }
272         }
273         return serviceClasses;
274     }

276     private void scanJavaFilesRecursive( File sourceDir, JavaProjectBuilder builder, List<JavaClass> serviceClasses )
277         throws IOException
278     {
279         File[] children = sourceDir.listFiles();
280         if ( children == null )
281         {
282             getLog().debug( "Directory does not exist: " + sourceDir );
283             return;
284         }
285         for ( File file : children )
286         {
287             if ( file.isDirectory() )
288             {
289                 scanJavaFilesRecursive( file, builder, serviceClasses );
290             }
291             else if ( file.getName().endsWith( ".java" ) )
292             {
293                 try
294                 {
295                     JavaSource source = builder.addSource( file );
296                     for ( JavaClass type : source.getClasses() )
297                     {
298                         boolean isService = isServiceClass( type );
299                         if ( isService )
300                         {
301                             serviceClasses.add( type );
302                         }
303                     }
304                 }
305                 catch ( Exception e )
306                 {
307                     getLog().debug( "Error parsing file: " + file, e );
308                 }
309             }
310         }
311 
312     }

314     private boolean isServiceClass( JavaClass type )
315     {
316         if ( this.classnamePattern.matcher( type.getName() ).matches() )
317         {
318             getLog().debug( "Class matches: " + type.getName() );
319             for ( JavaAnnotation annotation : type.getAnnotations() )
320             {
321                 if ( Path.class.getName().equals( annotation.getType().getFullyQualifiedName() ) )
322                 {
323                     getLog().info( "Found service: " + type.getName() );
324                     return true;
325                 }
326             }
327         }
328         return false;
329     }

331     private ClassLoader getProjectClassloader()
332         throws MojoExecutionException
333     {
334         if ( this.projectClassloader == null )
335         {
336             this.projectClassloader = new URLClassLoader( buildClasspathUrls(), this.getClass().getClassLoader() );
337         }
338         return this.projectClassloader;
339     }

341     private URL[] buildClasspathUrls()
342         throws MojoExecutionException
343     {
344         List<URL> urls = new ArrayList<URL>( this.runtimeClasspathElements.size() );
345         for ( String element : this.runtimeClasspathElements )
346         {
347             try
348             {
349                 URL url = new File( element ).toURI().toURL();
350                 urls.add( url );
351                 getLog().debug( "Adding to classloader: " + url.toString() );
352             }
353             catch ( MalformedURLException e )
354             {
355                 throw new MojoExecutionException( "Unable to access project dependency: " + element, e );
356             }
357         }
358         return urls.toArray( new URL[urls.size()] );
359     }
