139     /**
140      * The constructor.
141      *
142      * @param log the {@link Log}.
143      * @param project the {@link MavenProject}. May be <code>null</code> for testing.
144      * @param projectClassloader the {@link ClassLoader} to load byte-code.
145      * @param builder the {@link JavaProjectBuilder}.
146      * @param descriptor the pre-configured {@link ServicesDescriptor} template.
147      * @param introspectFields <code>true</code> to introspect beans using fields, <code>false</code> for getters.
148      */
149     public Analyzer( Log log, MavenProject project, ClassLoader projectClassloader, JavaProjectBuilder builder,
150                      ServicesDescriptor descriptor, boolean introspectFields )
151     {
152         super();
153         this.log = log;
154         this.project = project;
155         this.projectClassloader = projectClassloader;
156         this.reflectionUtil = ReflectionUtilImpl.getInstance();
157         this.annotationUtil = AnnotationUtilImpl.getInstance();
158         this.builder = builder;
159         this.descriptor = descriptor;
160         this.pojoDescriptorBuilderFactory = PojoDescriptorBuilderFactoryImpl.getInstance();
161         if ( introspectFields )
162         {
163             this.pojoDescriptorBuilder = this.pojoDescriptorBuilderFactory.createPrivateFieldDescriptorBuilder();
164         }
165         else
166         {
167             this.pojoDescriptorBuilder = this.pojoDescriptorBuilderFactory.createPublicMethodDescriptorBuilder();
168         }
169         this.javaDocHelper = new JavaDocHelper( this.projectClassloader, this.builder, this.descriptor.getJavadocs() );
170         if ( descriptor != null )
171         {
172             boolean hasFallback = false;
173             for ( ErrorDescriptor errorDescriptor : descriptor.getErrors() )
174             {
175                 if ( errorDescriptor.getMatch() == Match.assignable )
176                 {
177                     String errorName = errorDescriptor.getErrorName();
178                     Class<?> errorClass;
179                     try
180                     {
181                         errorClass = projectClassloader.loadClass( errorName );
182                     }
183                     catch ( Exception e )
184                     {
185                         log.warn( "Failed to load error class '" + errorName + "':" + e.getMessage(), e );
186                         errorClass = UnknownError.class;
187                     }
188                     errorDescriptor.setErrorClass( errorClass );
189                 }
190                 else if ( errorDescriptor.getMatch() == Match.regex )
191                 {
192                     String errorName = errorDescriptor.getErrorName();
193                     if ( errorName.isEmpty() || errorName.equals( "." ) || errorName.equals( ".*" ) )
194                     {
195                         hasFallback = true;
196                     }
197                 }
198             }
199             if ( !hasFallback )
200             {
201                 ErrorDescriptor errorDescriptor = new ErrorDescriptor();
202                 errorDescriptor.setErrorName( "" );
203                 descriptor.getErrors().add( errorDescriptor );
204             }
205         }
206     }

214     /**
215      * Does the actualy analysis.
216      *
217      * @param serviceClasses the {@link List} of service classes.
218      * @return the {@link ServicesDescriptor}.
219      * @throws Exception if something goes wrong.
220      */
221     public ServicesDescriptor createServicesDescriptor( List<JavaClass> serviceClasses )
222         throws Exception
223     {
224         if ( this.descriptor == null )
225         {
226             this.descriptor = new ServicesDescriptor();
227         }
228         InfoDescriptor info = this.descriptor.getInfo();
229         if ( info == null )
230         {
231             info = new InfoDescriptor();
232             this.descriptor.setInfo( info );
233         }
234         createInfoDescriptor( info );
235         Set<String> descriptorSchemes = this.descriptor.getSchemes();
236         if ( descriptorSchemes.isEmpty() )
237         {
238             descriptorSchemes.add( Descriptor.SCHEME_HTTPS );
239         }
240         for ( JavaClass type : serviceClasses )
241         {
242             ServiceDescriptor service = createServiceDescriptor( type );
243             this.descriptor.getServices().add( service );
244         }
245         return this.descriptor;
246     }

248     protected ServiceDescriptor createServiceDescriptor( JavaClass sourceType )
249         throws Exception
250     {
251         getLog().info( "Analyzing " + sourceType.getName() );
252         ServiceDescriptor serviceDescriptor = new ServiceDescriptor();
253         serviceDescriptor.setName( sourceType.getName() );
254         Class<?> byteClass;
255         try
256         {
257             byteClass = this.projectClassloader.loadClass( sourceType.getFullyQualifiedName() );
258         }
259         catch ( ClassNotFoundException e )
260         {
261             throw new MavenReportException( "Failed to load class " + sourceType.getName()
262                 + " - did you forget to invoke compile (e.g. mvn verify site)?", e );
263         }
264         Path serviceBasePath = byteClass.getAnnotation( Path.class );
265         if ( serviceBasePath != null )
266         {
267             serviceDescriptor.setBasePath( serviceBasePath.value() );
268         }
269         GenericType<?> byteType = this.reflectionUtil.createGenericType( byteClass );
270         serviceDescriptor.setJavaType( new JType( byteType, sourceType, this.reflectionUtil, this.javaDocHelper ) );
271         serviceDescriptor.setDescription( this.javaDocHelper.parseJavaDoc( sourceType, byteType,
272                                                                            sourceType.getComment() ) );
273         Consumes consumes = this.annotationUtil.getTypeAnnotation( byteClass, Consumes.class );
274         addConsumes( serviceDescriptor.getConsumes(), consumes );
275         Produces produces = this.annotationUtil.getTypeAnnotation( byteClass, Produces.class );
276         addProduces( serviceDescriptor.getProduces(), produces );
277         for ( Method byteMethod : byteClass.getMethods() )
278         {
279             getLog().debug( "Analyzing method " + byteMethod.toString() );
280             OperationDescriptor operationDescriptor = createOperationDescriptor( serviceDescriptor, byteMethod );
281             if ( operationDescriptor != null )
282             {
283                 getLog().debug( "Method has been detected as service operation." );
284                 serviceDescriptor.getOperations().add( operationDescriptor );
285             }
286         }
287         Collections.sort( serviceDescriptor.getOperations() );
288         return serviceDescriptor;
289     }

291     protected InfoDescriptor createInfoDescriptor( InfoDescriptor info )
292     {
293         if ( this.project == null )
294         {
295             return info;
296         }
297         if ( info.getVersion() == null )
298         {
299             info.setVersion( this.project.getVersion() );
300         }
301         if ( info.getTitle() == null )
302         {
303             info.setTitle( this.project.getArtifactId() );
304         }
305         if ( info.getDescription() == null )
306         {
307             info.setDescription( this.project.getDescription() );
308         }
309         ContactDescriptor contact = info.getContact();
310         if ( contact == null )
311         {
312             Organization organization = this.project.getOrganization();
313             if ( organization != null )
314             {
315                 contact = new ContactDescriptor();
316                 contact.setUrl( organization.getUrl() );
317                 contact.setName( organization.getName() );
318                 info.setContact( contact );
319             }
320             else
321             {
322                 String url = Util.getTrimmed( this.project.getUrl() );
323                 if ( !url.isEmpty() )
324                 {
325                     contact = new ContactDescriptor();
326                     contact.setUrl( url );
327                     String name = Util.getTrimmed( this.project.getName() );
328                     if ( name.isEmpty() )
329                     {
330                         name = Util.getTrimmed( this.project.getArtifactId() );
331                     }
332                     contact.setName( name );
333                     info.setContact( contact );
334                 }
335                 else
336                 {
337                     List<Developer> developers = this.project.getDevelopers();
338                     if ( ( developers != null ) && ( !developers.isEmpty() ) )
339                     {
340                         contact = new ContactDescriptor();
341                         Developer developer = developers.get( 0 );
342                         contact.setUrl( Util.getTrimmed( developer.getUrl() ) );
343                         contact.setEmail( Util.getTrimmed( developer.getEmail() ) );
344                         contact.setName( Util.getTrimmed( developer.getName() ) );
345                         info.setContact( contact );
346                     }
347                 }
348             }
349         }
350         LicenseDescriptor serviceLicense = info.getLicense();
351         if ( serviceLicense == null )
352         {
353             List<org.apache.maven.model.License> licenses = this.project.getLicenses();
354             if ( ( licenses != null ) && ( !licenses.isEmpty() ) )
355             {
356                 serviceLicense = new LicenseDescriptor();
357                 org.apache.maven.model.License projectLicense = licenses.get( 0 );
358                 serviceLicense.setName( projectLicense.getName() );
359                 serviceLicense.setUrl( projectLicense.getUrl() );
360                 info.setLicense( serviceLicense );
361             }
362         }
363         return info;
364     }

392     protected OperationDescriptor createOperationDescriptor( ServiceDescriptor serviceDescriptor, Method byteMethod )
393     {
394         Method annotatedParentMethod = byteMethod;
395         Path methodPath = null;
396         do
397         {
398             methodPath = annotatedParentMethod.getAnnotation( Path.class );
399             if ( methodPath == null )
400             {
401                 annotatedParentMethod = this.reflectionUtil.getParentMethod( annotatedParentMethod );
402                 if ( annotatedParentMethod == null )
403                 {
404                     return null;
405                 }
406             }
407         }
408         while ( methodPath == null );
409 
410         OperationDescriptor operationDescriptor = new OperationDescriptor();
411         operationDescriptor.setPath( methodPath.value() );
412 
413         if ( this.annotationUtil.getMethodAnnotation( byteMethod, Deprecated.class ) != null )
414         {
415             operationDescriptor.setDeprecated( true );
416         }
417         JMethod method = new JMethod( byteMethod, serviceDescriptor.getJavaType(), annotatedParentMethod );
418         operationDescriptor.setJavaMethod( method );
419         operationDescriptor.setDescription( method.getComment() );
420 
421         Set<String> consumes = operationDescriptor.getConsumes();
422         addConsumes( consumes, annotatedParentMethod.getAnnotation( Consumes.class ) );
423         if ( consumes.isEmpty() )
424         {
425             consumes.addAll( serviceDescriptor.getConsumes() );
426         }
427         Set<String> produces = operationDescriptor.getProduces();
428         addProduces( produces, annotatedParentMethod.getAnnotation( Produces.class ) );
429         if ( produces.isEmpty() )
430         {
431             produces.addAll( serviceDescriptor.getProduces() );
432         }
433 
434         operationDescriptor.setHttpMethod( createHttpMethodDescriptor( method ) );
435 
436         // parameters
437         for ( JParameter parameter : method.getParameters() )
438         {
439             ParameterDescriptor parameterDescriptor =
440                 createParameterDescriptor( serviceDescriptor, operationDescriptor, parameter );
441             if ( parameterDescriptor != null )
442             {
443                 operationDescriptor.getParameters().add( parameterDescriptor );
444             }
445         }
446 
447         // responses
448         ResponseDescriptor responseSuccess =
449             createResponseDescriptor( serviceDescriptor, operationDescriptor, method.getReturns(), false );
450         operationDescriptor.getResponses().add( responseSuccess );
451 
452         for ( JException exception : method.getExceptions() )
453         {
454             ResponseDescriptor response =
455                 createResponseDescriptor( serviceDescriptor, operationDescriptor, exception, true );
456             operationDescriptor.getResponses().add( response );
457         }
458 
459         for ( ErrorDescriptor errorDescriptor : this.descriptor.getErrors() )
460         {
461             if ( errorDescriptor.getMatch() == Match.always )
462             {
463                 JElement exception = new JException( this.reflectionUtil.createGenericType( Throwable.class ), null,
464                                                      errorDescriptor.getComment() );
465                 ResponseDescriptor response =
466                     createResponseDescriptor( serviceDescriptor, operationDescriptor, exception, true );
467                 operationDescriptor.getResponses().add( response );
468             }
469         }
470 
471         return operationDescriptor;
472     }

474     protected ParameterDescriptor createParameterDescriptor( ServiceDescriptor serviceDescriptor,
475                                                              OperationDescriptor operationDescriptor,
476                                                              JParameter parameter )
477     {
478         ParameterDescriptor parameterDescriptor = new ParameterDescriptor();
479         parameterDescriptor.setJavaParameter( parameter );
480         parameterDescriptor.setName( parameter.getName() );
481         parameterDescriptor.setDescription( parameter.getComment() );
482         String location = Descriptor.LOCATION_BODY;
483         boolean required = false;
484         for ( Annotation annotation : parameter.getByteAnnotations() )
485         {
486             if ( annotation instanceof QueryParam )
487             {
488                 QueryParam queryParam = (QueryParam) annotation;
489                 location = Descriptor.LOCATION_QUERY;
490                 parameterDescriptor.setName( queryParam.value() );
491             }
492             else if ( annotation instanceof HeaderParam )
493             {
494                 HeaderParam headerParam = (HeaderParam) annotation;
495                 location = Descriptor.LOCATION_HEADER;
496                 parameterDescriptor.setName( headerParam.value() );
497             }
498             else if ( annotation instanceof PathParam )
499             {
500                 PathParam pathParam = (PathParam) annotation;
501                 location = Descriptor.LOCATION_PATH;
502                 parameterDescriptor.setName( pathParam.value() );
503             }
504             else if ( annotation instanceof FormParam )
505             {
506                 FormParam formParam = (FormParam) annotation;
507                 location = Descriptor.LOCATION_FORM_DATA;
508                 parameterDescriptor.setName( formParam.value() );
509             }
510             else if ( annotation instanceof CookieParam )
511             {
512                 CookieParam cookieParam = (CookieParam) annotation;
513                 location = Descriptor.LOCATION_COOKIE;
514                 parameterDescriptor.setName( cookieParam.value() );
515             }
516             else if ( annotation instanceof Context )
517             {
518                 if ( UriInfo.class.isAssignableFrom( parameter.getByteType().getAssignmentClass() ) )
519                 {
520                     location = "query/path";
521                 }
522                 else
523                 {
524                     return null;
525                 }
526             }
527             else if ( annotation instanceof DefaultValue )
528             {
529                 DefaultValue defaultValue = (DefaultValue) annotation;
530                 parameterDescriptor.setDefaultValue( defaultValue.value() );
531             }
532             else if ( annotation instanceof NotNull )
533             {
534                 required = true;
535             }
536             else if ( annotation instanceof Mandatory )
537             {
538                 required = true;
539             }
540         }
541         parameterDescriptor.setLocation( location );
542         parameterDescriptor.setRequired( required );
543         JavaScriptType javaScriptType = getJavaScriptType( parameter.getByteType(), false );
544         parameterDescriptor.setJavaScriptType( javaScriptType.getName() );
545         parameterDescriptor.setExample( createExample( operationDescriptor, javaScriptType, parameter ) );
546         return parameterDescriptor;
547     }

549     protected ResponseDescriptor createResponseDescriptor( ServiceDescriptor serviceDescriptor,
550                                                            OperationDescriptor operationDescriptor,
551                                                            JElement javaElement, boolean error )
552     {
553         String reason = error ? RESPONSE_REASON_ERROR : RESPONSE_REASON_SUCCESS;
554         String example = null;
555         ResponseDescriptor response = new ResponseDescriptor();
556         GenericType<?> byteReturnType = javaElement.getByteType();
557         String description = javaElement.getComment();
558         if ( error )
559         {
560             String statusCode = Descriptor.STATUS_CODE_INTERNAL_SERVER_ERROR;
561             for ( ErrorDescriptor errorDescriptor : this.descriptor.getErrors() )
562             {
563                 boolean matches = isMatchingError( byteReturnType, errorDescriptor );
564                 if ( matches )
565                 {
566                     statusCode = errorDescriptor.getStatusCode();
567                     boolean isXml = Util.containsSubstring( operationDescriptor.getProduces(), "xml" );
568                     if ( isXml )
569                     {
570                         example = errorDescriptor.getXmlExample();
571                     }
572                     else
573                     {
574                         example = errorDescriptor.getJsonExample();
575                     }
576                     break;
577                 }
578             }
579             response.setStatusCode( statusCode );
580         }
581         else
582         {
583             if ( byteReturnType.getRetrievalClass() == void.class )
584             {
585                 response.setStatusCode( Descriptor.STATUS_CODE_NO_CONTENT );
586                 if ( Util.isEmpty( description ) )
587                 {
588                     description = DESCRIPTION_VOID;
589                 }
590             }
591             else
592             {
593                 response.setStatusCode( Descriptor.STATUS_CODE_SUCCESS );
594             }
595         }
596         response.setDescription( description );
597         response.setReason( reason );
598         response.setJavaElement( javaElement );
599         JavaScriptType javaScriptType = getJavaScriptType( byteReturnType, true );
600         response.setJavaScriptType( javaScriptType.getName() );
601         if ( example == null )
602         {
603             example = createExample( operationDescriptor, javaScriptType, javaElement );
604         }
605         response.setExample( example );
606         return response;
607     }

609     private boolean isMatchingError( GenericType<?> byteReturnType, ErrorDescriptor errorDescriptor )
610     {
611         Class<?> byteClass = byteReturnType.getRetrievalClass();
612         Match match = errorDescriptor.getMatch();
613         switch ( match )
614         {
615             case regex:
616                 Matcher matcher = errorDescriptor.getErrorNamePattern().matcher( byteClass.getName() );
617                 return matcher.matches();
618             case assignable:
619                 return errorDescriptor.getErrorClass().isAssignableFrom( byteClass );
620             case always:
621                 // yes, this is not a bug.
622                 // always means it will always produce an error result but independent of a specific exception
623                 return ( byteClass == Throwable.class );
624             default:
625                 throw new IllegalCaseException( Match.class, match );
626         }
627     }

629     private String createHttpMethodDescriptor( JMethod method )
630     {
631         Method byteMethod = method.getByteMethod();
632         if ( byteMethod.isAnnotationPresent( GET.class ) )
633         {
634             return Descriptor.HTTP_METHOD_GET;
635         }
636         else if ( byteMethod.isAnnotationPresent( PUT.class ) )
637         {
638             return Descriptor.HTTP_METHOD_PUT;
639         }
640         else if ( byteMethod.isAnnotationPresent( POST.class ) )
641         {
642             return Descriptor.HTTP_METHOD_POST;
643         }
644         else if ( byteMethod.isAnnotationPresent( DELETE.class ) )
645         {
646             return Descriptor.HTTP_METHOD_DELETE;
647         }
648         else if ( byteMethod.isAnnotationPresent( OPTIONS.class ) )
649         {
650             return Descriptor.HTTP_METHOD_OPTIONS;
651         }
652         else if ( byteMethod.isAnnotationPresent( HEAD.class ) )
653         {
654             return Descriptor.HTTP_METHOD_HEAD;
655         }
656         else
657         {
658             getLog().warn( "Service method " + method + " is missing JAX-RS annotation for HTTP method!" );
659             return null;
660         }
661     }

668     private JavaScriptType getJavaScriptType( GenericType<?> byteType, boolean retrieval, int recursion )
669     {
670         Class<?> clazz;
671         if ( retrieval )
672         {
673             clazz = byteType.getRetrievalClass();
674         }
675         else
676         {
677             clazz = byteType.getAssignmentClass();
678         }
679         Class<?> byteClass = this.reflectionUtil.getNonPrimitiveType( clazz );
680         if ( Number.class.isAssignableFrom( byteClass ) )
681         {
682             NumberType<? extends Number> numberType = MathUtilImpl.getInstance().getNumberType( byteClass );
683             if ( ( numberType == null ) || numberType.isDecimal() )
684             {
685                 return JavaScriptType.NUMBER;
686             }
687             else
688             {
689                 return JavaScriptType.INTEGER;
690             }
691         }
692         else if ( byteClass.isArray() || ( Collection.class.isAssignableFrom( byteClass ) ) )
693         {
694             return JavaScriptType.ARRAY;
695         }
696         else if ( Boolean.class.equals( byteClass ) )
697         {
698             return JavaScriptType.BOOLEAN;
699         }
700         else if ( Void.class.isAssignableFrom( byteClass ) )
701         {
702             return JavaScriptType.VOID;
703         }
704         else if ( CharSequence.class.isAssignableFrom( byteClass ) )
705         {
706             return JavaScriptType.STRING;
707         }
708         else if ( SimpleDatatype.class.isAssignableFrom( byteClass ) )
709         {
710             if ( recursion <= 2 )
711             {
712                 Type typeVariable = SimpleDatatype.class.getTypeParameters()[0];
713                 GenericType<?> datatype = this.reflectionUtil.createGenericType( typeVariable, byteType );
714                 return getJavaScriptType( datatype, retrieval, recursion + 1 );
715             }
716         }
717         else if ( byteClass.isEnum() )
718         {
719             return JavaScriptType.STRING;
720         }
721         else if ( isDate( byteClass ) )
722         {
723             return JavaScriptType.DATE;
724         }
725         else if ( Type.class.isAssignableFrom( byteClass ) )
726         {
727             return JavaScriptType.TYPE;
728         }
729         return JavaScriptType.OBJECT;
730     }

732     private boolean isDate( Class<?> byteClass )
733     {
734         if ( Date.class.isAssignableFrom( byteClass ) )
735         {
736             return true;
737         }
738         else if ( Calendar.class.isAssignableFrom( byteClass ) )
739         {
740             return true;
741         }
742         else
743         {
744             String className = byteClass.getName();
745             if ( "java.time.LocalDate".equals( className ) )
746             {
747                 return true;
748             }
749             else if ( "java.time.LocalDateTime".equals( className ) )
750             {
751                 return true;
752             }
753             else if ( "java.time.Instant".equals( className ) )
754             {
755                 return true;
756             }
757             else if ( "java.time.OffsetDateTime".equals( className ) )
758             {
759                 return true;
760             }
761             else if ( "java.time.ZonedDateTime".equals( className ) )
762             {
763                 return true;
764             }
765         }
766         return false;
767     }

769     protected String createExample( OperationDescriptor operationDescriptor, JavaScriptType javaScriptType,
770                                     JElement element )
771     {
772         if ( javaScriptType == JavaScriptType.VOID )
773         {
774             return null;
775         }
776         boolean retrieval = ( element instanceof JReturn );
777         GenericType<?> byteType = element.getByteType();
778         Class<?> byteClass;
779         if ( retrieval )
780         {
781             byteClass = byteType.getRetrievalClass();
782         }
783         else
784         {
785             byteClass = byteType.getAssignmentClass();
786         }
787         StringBuilder buffer = new StringBuilder();
788         createExample( javaScriptType, byteType, byteClass, "", buffer, new HashSet<Class<?>>(), retrieval );
789         return buffer.toString();
790     }

794     {

805         createExample( javaScriptType, byteType, byteClass, indent, buffer, visitedClassSet, retrieval );

808     private void createExample( JavaScriptType javaScriptType, GenericType<?> byteType, Class<?> byteClass,
809                                 String indent, StringBuilder buffer, Set<Class<?>> visitedClassSet, boolean retrieval )
810     {
811         if ( javaScriptType == JavaScriptType.ARRAY )
812         {
813             GenericType<?> componentType = byteType.getComponentType();
814             if ( componentType != null )
815             {
816                 buffer.append( '[' );
817                 createExample( componentType, indent, buffer, visitedClassSet, retrieval );
818                 buffer.append( ']' );
819                 return;
820             }
821         }
822         else if ( javaScriptType == JavaScriptType.OBJECT )
823         {
824             boolean added = true;
825             if ( !Datatype.class.isAssignableFrom( byteClass ) )
826             {
827                 added = visitedClassSet.add( byteClass );
828             }
829             if ( byteClass == Object.class )
830             {
831                 added = false;
832             }
833             if ( added )
834             {
835                 buffer.append( '{' );
836                 String childIndent = indent + "  ";
837                 buffer.append( '\n' );
838                 buffer.append( childIndent );
839                 if ( Map.class.isAssignableFrom( byteClass ) )
840                 {
841                     createExampleForMap( byteType, buffer, visitedClassSet, retrieval, childIndent );
842                 }
843                 else
844                 {
845                     createExampleForBean( byteType, byteClass, buffer, visitedClassSet, retrieval, childIndent );
846                 }
847                 buffer.append( '\n' );
848                 buffer.append( indent );
849                 buffer.append( '}' );
850                 return;
851             }
852         }
853         buffer.append( javaScriptType.getExample() );
854     }

856     private void createExampleForBean( GenericType<?> byteType, Class<?> byteClass, StringBuilder buffer,
857                                        Set<Class<?>> visitedClassSet, boolean retrieval, String childIndent )
858     {
859         PojoDescriptor<?> pojoDescriptor = this.pojoDescriptorBuilder.getDescriptor( byteType );
860         boolean repeating = false;
861         List<? extends PojoPropertyDescriptor> propertyDescriptors =
862             new ArrayList<PojoPropertyDescriptor>( pojoDescriptor.getPropertyDescriptors() );
863         Comparator<PojoPropertyDescriptor> comparator = new Comparator<PojoPropertyDescriptor>()
864         {
865             public int compare( PojoPropertyDescriptor o1, PojoPropertyDescriptor o2 )
866             {
867                 return o1.getName().compareTo( o2.getName() );
868             }
869         };
870         Collections.sort( propertyDescriptors, comparator );
871         for ( PojoPropertyDescriptor propertyDescriptor : propertyDescriptors )
872         {
873             if ( !propertyDescriptor.getName().equals( "class" ) )
874             {
875                 PojoPropertyAccessorNonArg getter =
876                     propertyDescriptor.getAccessor( PojoPropertyAccessorNonArgMode.GET );
877                 if ( getter != null )
878                 {
879                     if ( repeating )
880                     {
881                         buffer.append( ",\n" );
882                         buffer.append( childIndent );
883                     }
884                     repeating = true;
885                     String propertyName = propertyDescriptor.getName();
886                     buffer.append( '"' );
887                     buffer.append( propertyName );
888                     buffer.append( "\" = " );
889                     createExample( getter.getPropertyType(), childIndent, buffer, visitedClassSet, retrieval );
890                 }
891             }
892         }
893     }

895     private void createExampleForMap( GenericType<?> byteType, StringBuilder buffer, Set<Class<?>> visitedClassSet,
896                                       boolean retrieval, String childIndent )
897     {
898         buffer.append( "\"&lt;key&gt;\" = " );
899         GenericType<?> componentType = byteType.getComponentType();
900         if ( componentType == null )
901         {
902             buffer.append( "..." );
903         }
904         else
905         {
906             createExample( componentType, childIndent, buffer, visitedClassSet, retrieval );
907             buffer.append( '\n' );
908             buffer.append( childIndent );
909             buffer.append( ", ..." );
910         }
911     }
